void HostelManagementSystem::del_student()
{
			string line;
		    string name;
		    bool found = false;
		    cout << "\t\tEnter the name to be deleted: ";
		    cin >> name;
		    ifstream myfile("Student.txt");
		    ofstream temp("temp.txt");
		    if(myfile.is_open() && temp.is_open())
		    {
		        while(getline(myfile, line))
		        {
		            if(line.find(name) == string::npos)
		            {
		                temp << line << endl;
		            }
		            else
		            {
		                found = true;
		            }
		        }
		    }
		    else
		    {
		        cout << "\t\tUnable to open file";
		    }
		    myfile.close();
		    temp.close();
		    remove("Student.txt");
		    rename("temp.txt", "Student.txt");
		    if(found)
		    {
		        cout << "\t\tRecord deleted successfully";
		    }
		    else
		    {
		        cout << "\t\tRecord not found"<<endl;
		    }
		    cout<<endl;
		    cout<<"\t\tPRESS ANY KEY TO GO BACK TO HOMEPAGE."<<endl;
        	_getch();
}

{

			cin>>NAME;
			ifstream fi("student.txt");
			if(! fi.is_open())
			{
				cout<<"file is not open";
			}
			else
			{
				string line;
				bool found = false;
			while (getline(fi, line, '\n'))
			{
				if (line.find(NAME) != string::npos)
				{
					cout<<"\t\t"<<line <<endl;
					found = true;
					break;
				}
			}
		if (found == false)
			cout << NAME << " is not found!"<<endl;
			fi.close();
			}
			cout<<endl<<endl;

        	_getch();
}
